CTable Basic Example

This is a basic example of using CTable with simple text columns.

// Basic configuration
const tableConfig = {
  Columns: [
    { prop: 'id', label: 'ID', width: 80 },
    { prop: 'name', label: 'Name', width: 150 },
    { prop: 'email', label: 'Email', width: 200 },
    { prop: 'role', label: 'Role', width: 120 }
  ],
  rowStyle: { height: 35 },
  headRowStyle: { backgroundColor: '#4a90e2', height: 40 }
};

const table = new CTable('table-container', tableConfig);
table.setTableData([
  { id: 1, name: 'John Doe', email: 'john@example.com', role: 'Developer' },
  { id: 2, name: 'Jane Smith', email: 'jane@example.com', role: 'Designer' },
  { id: 3, name: 'Bob Johnson', email: 'bob@example.com', role: 'Manager' },
  { id: 4, name: 'Alice Williams', email: 'alice@example.com', role: 'Analyst' },
  { id: 5, name: 'Charlie Brown', email: 'charlie@example.com', role: 'Tester' }
]);
