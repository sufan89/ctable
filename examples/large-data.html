CTable Large Data Example

This example demonstrates CTable's performance with large datasets (10,000+ rows). CTable uses virtual scrolling to efficiently render only the visible rows.

Click a button to load data
// Generating large dataset
function generateData(count) {
  const data = [];
  const roles = ['Developer', 'Designer', 'Manager', 'Analyst', 'Tester'];
  const departments = ['Engineering', 'Design', 'Management', 'Finance', 'QA', 'HR', 'Marketing'];

  for (let i = 0; i < count; i++) {
    data.push({
      id: i + 1,
      name: `User ${i + 1}`,
      email: `user${i + 1}@example.com`,
      role: roles[i % roles.length],
      department: departments[i % departments.length],
      salary: Math.floor(Math.random() * 100000) + 50000
    });
  }
  return data;
}
